# 실행방법
# 1. 명령프롬프트를 실행한다.
# 2. python gugudan.py 6
# 3. 실행결과: 구구단이 출력되도, gugudan.txt파일이 생성된다.
# 4. 메모장이나 텍스트에디터로 확인해 본다.
import sys
x,y = (2,2)
with open('gugudan.txt', 'w') as f:
    while True:
        if x > int(sys.argv[1]):
            break
        while True:
            if y > int(sys.argv[1]):
                break
            msg = f'{x} x {y} = {x*y} '
            print(msg, end = '')
            f.write(msg)
            y += 1
        y = 2
        x += 1
        print('')
        f.write('\n')
  #+END_SRC
